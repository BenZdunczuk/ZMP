#define ROTATE_SPEED 5
/*
* Przykładowy zestaw poleceń
*/
Set Podstawa1 0 0 0 0 0 45 // Polozenie obiektu A
Set Podstawa1.Ramie1 0 0 0 0 0 0 // Polozenie obiektu B
Set Podstawa1.Ramie1.Ramie2 0 0 0 0 0 0
Pause 2000
Rotate Podstawa1.Ramie1 OY ROTATE_SPEED 90
Pause 2000 /* Zawieszenie dzialania na 5 sek. */
Move Podstawa1 1 2
Pause 2000 /* Zawieszenie dzialania na 3 sek. */
Begin_Parallel_Actions
Rotate Podstawa1.Ramie1 OY ROTATE_SPEED 45
Pause 2000 /* Zawieszenie dzialania na 3 sek. */
Rotate Podstawa1.Ramie1.Ramie2 OY 36 360
Pause 2000 /* Zawieszenie dzialania na 3 sek. */
End_Parallel_Actions
Rotate Podstawa1.Ramie1.Ramie2 OX 36 360
Pause 2000 /* Zawieszenie dzialania na 3 sek. */
